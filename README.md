def is_isomorphic(s, t):
    if len(s) != len(t):
        return False

    map_s_to_t = {}
    map_t_to_s = {}

    for char_s, char_t in zip(s,t):
        if char_s in map_s_to_t and map_s_to_t[char_s] != char_t:
            return False

        if char_t in map_t_to_s and map_t_to_s[char_t] != char_s:
            return False

        map_s_to_t[char_s] = char_t
        map_t_to_s[char_t] = char_s

    return True
 Идея решения:
Используются два словаря: первый хранит соответствие символов из строки s символам строки t, второй — обратное соответствие. Это позволяет проверить два условия изоморфности: одинаковые символы первой строки должны переходить в одинаковые символы второй строки, а разные символы первой строки не должны переходить в один и тот же символ второй строки.
Сложность по времени: O(n), где n — длина строк, так как выполняется один проход по символам строк.
Сложность по памяти: O(n), так как в худшем случае в словарях хранится до n различных соответствий.
Оптимальность:
Решение является оптимальным по времени, поскольку для проверки изоморфности необходимо как минимум один раз просмотреть все символы строк
2.2) def missing_number(nums):
    # 1. Определяем, какое максимальное число должно быть в ряду.
    # Так как одного числа не хватает, n — это длина списка + 1.
    # В примере [1, 2, 3, 4, 5, 6, 8, 9, 10, 11] длина 10, значит n = 11.
    n = len(nums) + 1
    
    # 2. Вычисляем "идеальную" сумму всех чисел от 1 до n по формуле.
    # Используем целочисленное деление // 2, чтобы результат был целым числом.
    expected_sum = (n * (n + 1)) // 2
    
    # 3. Считаем сумму чисел, которые у нас есть в списке.
    actual_sum = sum(nums)
    
    # 4. Разница — это и есть наше потерянное число.
    return expected_sum - actual_sum

# Проверка примера:
nums = [1, 2, 3, 4, 5, 6, 8, 9, 10, 11]
print(missing_number(nums))  # Выведет: 7
